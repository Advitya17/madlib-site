00106     rel_output      VARCHAR,
00107     rel_source      REGCLASS,
00108     col_row         VARCHAR,
00109     col_column      VARCHAR,
00110     col_value       VARCHAR,
00111     row_dim         INTEGER,            /*+ DEFAULT SELECT max(col_row) FROM rel_source */
00112     column_dim      INTEGER,            /*+ DEFAULT SELECT max(col_column) FROM rel_source */
00113     max_rank        INTEGER,            /*+ DEFAULT 20 */
00114     stepsize        DOUBLE PRECISION,   /*+ DEFAULT 0.01 */
00115     scale_factor    DOUBLE PRECISION,   /*+ DEFAULT 0.1 */
00116     num_iterations  INTEGER,            /*+ DEFAULT 10 */
00117     tolerance       DOUBLE PRECISION)   /*+ DEFAULT 0.0001 */
00118 RETURNS INTEGER AS $$
00119 DECLARE
00120     iteration_run   INTEGER;
00121     model_id        INTEGER;
00122     rmse            DOUBLE PRECISION;
00123     old_messages    VARCHAR;
00124 BEGIN
00125     RAISE NOTICE 'Matrix % to be factorized: % x %', rel_source, row_dim, column_dim;
00126 
00127     -- We first setup the argument table. Rationale: We want to avoid all data
00128     -- conversion between native types and Python code. Instead, we use Python
00129     -- as a pure driver layer.
00130     old_messages :=
00131         (SELECT setting FROM pg_settings WHERE name = 'client_min_messages');
00132     EXECUTE 'SET client_min_messages TO warning';
00133     PERFORM MADLIB_SCHEMA.create_schema_pg_temp();
00134     -- Unfortunately, the EXECUTE USING syntax is only available starting
00135     -- PostgreSQL 8.4:
00136     -- http://www.postgresql.org/docs/8.4/static/plpgsql-statements.html#PLPGSQL-STATEMENTS-EXECUTING-DYN
00137     -- We therefore have to emulate.
00138     PERFORM MADLIB_SCHEMA.internal_execute_using_lmf_igd_args($sql$
00139         DROP TABLE IF EXISTS pg_temp._madlib_lmf_igd_args;
00140         CREATE TABLE pg_temp._madlib_lmf_igd_args AS
00141         SELECT 
00142             $1 AS row_dim, 
00143             $2 AS column_dim,
00144             $3 AS max_rank, 
00145             $4 AS stepsize,
00146             $5 AS scale_factor, 
00147             $6 AS num_iterations, 
00148             $7 AS tolerance;
00149         $sql$,
00150         row_dim, column_dim, max_rank, stepsize,
00151         scale_factor, num_iterations, tolerance);
00152     EXECUTE 'SET client_min_messages TO ' || old_messages;
00153 
00154     -- Perform acutal computation.
00155     -- Unfortunately, Greenplum and PostgreSQL <= 8.2 do not have conversion
00156     -- operators from regclass to varchar/text.
00157     iteration_run := MADLIB_SCHEMA.internal_compute_lmf_igd(
00158             '_madlib_lmf_igd_args', '_madlib_lmf_igd_state',
00159             textin(regclassout(rel_source)), col_row, col_column, col_value);
00160 
00161     -- create result table if it does not exist
00162     BEGIN
00163         EXECUTE 'SELECT 1 FROM ' || rel_output || ' LIMIT 0';
00164     EXCEPTION
00165         WHEN undefined_table THEN
00166             EXECUTE '
00167             CREATE TABLE ' || rel_output || ' (
00168                 id                  serial,
00169                 flatten_matrix_u    DOUBLE PRECISION[],
00170                 flatten_matrix_v    DOUBLE PRECISION[],
00171                 rmse                DOUBLE PRECISION)';
00172     END;
00173 
00174     -- A work-around for GPDB not supporting RETURNING for INSERT
00175     -- We generate an id using nextval before INSERT
00176     EXECUTE '
00177     SELECT nextval(' || quote_literal(rel_output || '_id_seq') ||'::regclass)'
00178     INTO model_id;
00179 
00180     -- output model
00181     -- Retrieve result from state table and insert it
00182     EXECUTE '
00183     INSERT INTO ' || rel_output || '
00184     SELECT ' || model_id || ', (result).*
00185     FROM (
00186         SELECT MADLIB_SCHEMA.internal_lmf_igd_result(_state) AS result
00187         FROM _madlib_lmf_igd_state
00188         WHERE _iteration = ' || iteration_run || '
00189         ) subq';
00190 
00191     EXECUTE '
00192     SELECT rmse
00193     FROM ' || rel_output || '
00194     WHERE id = ' || model_id
00195     INTO rmse;
00196 
00197     -- return description
00198     RAISE NOTICE '
00199 Finished low-rank matrix factorization using incremental gradient 
00200  * table : % (%, %, %)
00201 Results:
00202  * RMSE = %
00203 Output:
00204  * view : SELECT * FROM % WHERE id = %',
00205     rel_source, col_row, col_column, col_value, rmse, rel_output, model_id;
00206     
00207     RETURN model_id;
00208 END;
00209 $$ LANGUAGE plpgsql VOLATILE;
